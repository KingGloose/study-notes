## 1.2 handleEvent

``` javascript
// 1. For Web Components
class MentionUser extends HTMLElement {
    static tagName = 'mention-user';
  
    static {
      customElements.define(MentionUser.tagName, this);
    }

    connectedCallback() {
      this.addEventListener('click', this);
    }
  
    handleEvent(event) {
      console.log(event.target.outerHTML);
    }
}

// 2. For Generic Class
class CanIUseHandleEvent {
  constructor() {
    document.addEventListener('click', this); // 这里填写 this 可以直接使用 handleEvent
  }
  
  handleEvent(event) {
    console.log(event.target.localName);
  }
}

new CanIUseHandleEvent();

```
